DECLARE
    a, b
BEGIN
    a := 0;
    b := 1;


    [ EQUALS ]
    IF 3 = 3 THEN
        IF 5 = 3 THEN
            WRITE a;
        ELSE
            WRITE b;
        ENDIF
    ELSE
        WRITE a;
    ENDIF


    [ NOT EQUALS ]
    IF 5 != 3 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ LESSER ]
    IF 3 < 5 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ GREATER ]
    IF 5 > 3 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ LEQ ] 
    IF 3 <= 3 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ GEQ ]
    IF 3 >= 3 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF

END
