DECLARE
    a, b
BEGIN
    a := 0;
    b := 1;

    [ EQUALS ]
    IF a = b THEN
        WRITE a;
    ELSE
        WRITE b;
    ENDIF

    IF a = 0 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ NOT EQUALS ]
    IF a != b THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF

    IF a != 0 THEN
        WRITE a;
    ELSE
        WRITE b;
    ENDIF


    [ GEQ ]
    IF a >= b THEN
        WRITE a;
    ELSE
        WRITE b;
    ENDIF

    IF a >= 0 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ LEQ ]
    IF a <= b THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF

    IF b <= 1 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ LESSER ]
    IF a < b THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF

    IF a < 10 THEN
        WRITE b;
    ELSE
        WRITE a;
    ENDIF


    [ GREATER ]
    IF a > b THEN
        WRITE a;
    ELSE
        WRITE b;
    ENDIF

    IF a > 10 THEN
        WRITE a;
    ELSE
        WRITE b;
    ENDIF

END
